016package com.googlecode.catchexception.internal;
017
018import org.mockito.cglib.proxy.MethodInterceptor;
019
020/**
021 * Creates proxies.
022 * 
023 * @author rwoo
024 * 
025 */
026interface ProxyFactory {
027
028    /**
029     * Create a proxy.
030     * 
031     * @param <T>
032     *            The type parameter makes some casts redundant.
033     * @param targetClass
034     *            the class the factory shall create a proxy for
035     * @param interceptor
036     *            the method interceptor that shall be applied to all method
037     *            calls.
038     * @return Returns the created proxy.
039     */
040    public <T> T createProxy(Class<?> targetClass, MethodInterceptor interceptor);
041
042}
